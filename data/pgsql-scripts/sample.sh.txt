#!/bin/bash
set -e

psql -v ON_ERROR_STOP=1 --username "$POSTGRES_USER" --dbname "$POSTGRES_DB" <<-EOSQL
    CREATE USER $SAMPLEDB_USER WITH PASSWORD '$SAMPLEDB_PASSWORD';
    CREATE DATABASE sampledb;
EOSQL


psql -v ON_ERROR_STOP=1 --username "$POSTGRES_USER" -d sampledb <<-EOSQL

    GRANT CREATE ON SCHEMA public TO $SAMPLEDB_USER;
    GRANT USAGE ON SCHEMA public TO $SAMPLEDB_USER;
    GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO $SAMPLEDB_USER;
    GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA public TO $SAMPLEDB_USER;

EOSQL
